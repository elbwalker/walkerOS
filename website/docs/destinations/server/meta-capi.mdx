
## Setup

```typescript
import { elb } from '@walkeros/collector';
import { destinationMeta } from '@walkeros/server-destination-meta';

elb('walker destination', destinationMeta, {
  settings: {
    accessToken: 'YOUR_ACCESS_TOKEN',
    pixelId: 'YOUR_PIXEL_ID',
  },
});
```
