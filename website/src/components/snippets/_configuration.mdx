export const configHrefs = {
  destination: '/docs/destinations#configuration',
  source: '/docs/sources#configuration',
  transformer: '/docs/transformers#configuration',
  store: '/docs/stores#configuration',
};

export const camelToSpaced = (key) =>
  key
    .replace(/([A-Z])/g, ' $1')
    .replace(/[_-]+/g, ' ')
    .trim()
    .toLowerCase();

export const stepTitle = (key, example) =>
  (example && example.title) || camelToSpaced(key);

This {props.type} uses the standard {props.type} config wrapper (consent, data, env, id, ...). For the shared fields see <a href={configHrefs[props.type]}>{props.type} configuration</a>. Package-specific fields live under `config.settings` and are listed below.

## Mapping

{props.data?.schemas?.mapping ? (
  <>
    <p>Per-event rules under <code>config.mapping</code>. For the standard rule fields (consent, condition, data, batch, name, policy) see <a href="/docs/mapping">mapping</a>.</p>
    <PropertyTable schema={props.data.schemas.mapping} />
  </>
) : (
  <p>This package does not define custom rule-level settings. For the standard rule fields (consent, condition, data, batch, name, policy) see <a href="/docs/mapping">mapping</a>.</p>
)}

export const visibleStepExamples = props.data?.examples?.step
  ? Object.entries(props.data.examples.step).filter(
      ([, example]) => example?.public !== false,
    )
  : [];

{visibleStepExamples.length > 0 && (
  <>
    <h2>Examples</h2>
    {visibleStepExamples.flatMap(([key, example]) => [
      <h3 key={`${key}-h`}>{stepTitle(key, example)}</h3>,
      <StepExample key={`${key}-e`} example={example} />,
    ])}
  </>
)}
